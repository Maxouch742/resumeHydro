    b = \cfrac{\sqrt{6}}{\pi} \cdot \sigma_{\overline{Q}_\text{mes}}  \qquad  \sigma_{\overline{Q}_\text{mes}} = b \cdot \cfrac{\pi}{\sqrt{6}}                  \\
\end{eqnarray*}
